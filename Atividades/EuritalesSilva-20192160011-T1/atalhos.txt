#include <stdio.h>

int main() {

  int idade_dia, dia, mes, ano;

  printf("Insira sua idade em dias:\n");
  scanf("%d",&idade_dia);

  ano = idade_dia/365;
  mes = ano*12;
  dia = idade_dia;
  printf("Sua idade em anos é:%d\n", ano);
  printf("Sua idade em meses é:%d\n", mes);
  printf("Sua idade em dias é:%d\n", dia);
  
  return 0;
}
